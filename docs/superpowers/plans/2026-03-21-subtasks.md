# Subtasks Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add subtask capability to issues — a nested endpoint, count annotations, a SubtaskList UI in the detail page, and a counter badge on Kanban cards.

**Architecture:** The existing `Issue.parent` FK and `Issue.Type.SUBTASK` are already in place. A new `GET/POST /issues/{id}/subtasks/` endpoint follows the exact pattern of existing nested resources (comments, relations). The frontend adds `useSubtasks`/`useCreateSubtask` hooks, a `SubtaskList` component, and a count badge on `IssueCard` in `BoardPage.tsx`.

**Tech Stack:** Django REST Framework, PostgreSQL, React 18, TypeScript, TanStack Query v5, Tailwind CSS, Vitest + Testing Library

**Spec:** `docs/superpowers/specs/2026-03-21-subtasks-design.md`

class IssueSubtaskListCreateView(generics.ListCreateAPIView):
  """GET/POST /issues/{issue_pk}/subtasks/"""
  permission_classes = [IsAuthenticated]
  pagination_class = None  # subtasks are bounded per parent; no pagination needed

  def _get_issue(self):
    return _get_issue(self.kwargs["issue_pk"], self.request.user)

  def get_serializer_class(self):
    # IssueSerializer and SubtaskSerializer are imported at module level (see top of file)
    if self.request.method == "POST":
      return IssueSerializer
    return SubtaskSerializer

  def get_queryset(self):
    return self._get_issue().sub_issues.select_related(
      "state", "assignee"
    ).order_by("sort_order")

  def perform_create(self, serializer):
    parent = self._get_issue()
    if parent.parent_id is not None:
      raise ValidationError("Subtarefas não podem ter filhos.")
    serializer.save(
      project=parent.project,
      parent=parent,
      type=Issue.Type.SUBTASK,
    )
```

export function useSubtasks(issueId: string) {
  return useQuery({
    queryKey: ['subtasks', issueId],
    queryFn: () => issueService.subtasks(issueId),
    enabled: !!issueId,
  })
}

export function useCreateSubtask() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: ({ issueId, data }: { issueId: string; data: CreateSubtaskDto }) => {
      const payload: Record<string, unknown> = {
        title: data.title,
        priority: data.priority ?? 'none',
      }
      if (data.stateId) payload.state = data.stateId
      if (data.assigneeId) payload.assignee = data.assigneeId
      if (data.description) payload.description = data.description
      if (Array.isArray(data.labelIds) && data.labelIds.length > 0) {
        payload.label_ids = data.labelIds
      }
      return issueService.createSubtask(issueId, payload)
    },
    onSuccess: (_, { issueId }) => {
      qc.invalidateQueries({ queryKey: ['subtasks', issueId] })
      qc.invalidateQueries({ queryKey: ['issue', issueId] })
    },
  })
}
```

```tsx
import { useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { Plus } from 'lucide-react'
import { useSubtasks } from '@/hooks/useIssues'
import { Avatar } from '@/components/ui/Avatar'
import { Button } from '@/components/ui/Button'
import { PriorityCapsule } from '@/components/ui/IssueCapsules'
import { IssueForm } from './IssueForm'
import type { Issue } from '@/types'

interface SubtaskListProps {
  projectId: string
  issueId: string
}

export function SubtaskList({ projectId, issueId }: SubtaskListProps) {
  const navigate = useNavigate()
  const { data: subtasks, isLoading, isError } = useSubtasks(issueId)
  const [addingSubtask, setAddingSubtask] = useState(false)

  return (
    <section aria-label="Subtarefas" className="mb-6">
      <div className="mb-2 flex items-center justify-between">
        <h3 className="text-sm font-semibold text-gray-700 dark:text-gray-300">
          Subtarefas ({isLoading ? '…' : (subtasks?.length ?? 0)})
        </h3>
        <Button
          variant="ghost"
          size="sm"
          onClick={() => setAddingSubtask(true)}
          aria-label="Adicionar subtarefa"
        >
          <Plus className="mr-1 h-3.5 w-3.5" aria-hidden="true" />
          Adicionar
        </Button>
      </div>

      {isLoading && (
        <div role="status" aria-label="Carregando subtarefas" className="py-2 text-xs text-gray-400 dark:text-gray-500">
          Carregando…
        </div>
      )}

      {isError && (
        <p className="py-2 text-xs text-gray-400 dark:text-gray-500">
          Não foi possível carregar subtarefas.
        </p>
      )}

      {!isLoading && !isError && subtasks?.length === 0 && (
        <p className="py-2 text-xs text-gray-400 dark:text-gray-500 italic">
          Sem subtarefas
        </p>
      )}

      {!isLoading && !isError && subtasks && subtasks.length > 0 && (
        <ul className="space-y-1" role="list">
          {subtasks.map((sub: Issue) => (
            <li
              key={sub.id}
              className="flex items-center gap-2 rounded-md px-2 py-1.5 text-sm hover:bg-gray-50 dark:hover:bg-gray-800 cursor-pointer"
              onClick={() =>
                navigate(`/projects/${projectId}/issues/${sub.id}`, {
                  state: { from: `/projects/${projectId}/issues/${issueId}` },
                })
              }
              role="listitem"
            >
              <span
                className="h-2 w-2 shrink-0 rounded-full"
                style={{ backgroundColor: sub.stateColor ?? '#ccc' }}
                aria-hidden="true"
              />
              <span className="shrink-0 font-mono text-xs text-gray-400 dark:text-gray-500">
                #{sub.sequenceId}
              </span>
              <span className="flex-1 truncate text-gray-900 dark:text-gray-100">
                {sub.title}
              </span>
              <PriorityCapsule priority={sub.priority} />
              {sub.assignee && (
                <Avatar
                  src={sub.assignee.avatarUrl}
                  name={sub.assignee.name}
                  size="xs"
                  className="shrink-0"
                />
              )}
            </li>
          ))}
        </ul>
      )}

      <IssueForm
        projectId={projectId}
        open={addingSubtask}
        onClose={() => setAddingSubtask(false)}
        parentIssueId={issueId}
        typeOverride="subtask"
      />
    </section>
  )
}
```
